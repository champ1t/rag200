Add onu to Huawei by manual
เขียนโดย Maru^^   
วันเสาร์ที่ 18 มกราคม 2014 เวลา 17:23 น.

#enable

#display board 0                       // ดู card ที่เป็น GPBC

#config

#display ont-lineprofile gpon all    // ดู line-profile

#display ont-srvprofile gpon all    // ดู service-profile

#display vlan all                         // ดู vlan

#display traffic table ip from-index 0   // ดู tx - rx traffic profile (สุ่ม index)

#config

#interface gpon (slot/port (0/5))

#display ont autofind (port-id(0-7))

#ont add (portid (0-7)) (onuid (0-127)) sn-auth (serial number) omci ont-lineprofile-id (ont-lineprofileid) ont-srvprofile-id (ont-srvprofileid) desc (Number)

#quit

----------- Add Internet ----------

#service-port vlan (vlan id) gpon (fr/sl/port) ont (onuid) gemport 1 multi-service user-vlan 10 rx-cttr (uploadid) tx-cttr (downloadid)

----------- Add Voice ----------

#service-port vlan (vlan-void id) gpon (fr/sl/port) ont (onuid) gemport 2 multi-service user-vlan 20 rx-cttr (uploadid) tx-cttr (downloadid)

----------- EX ----------

#ont add 2 129 sn-auth 48575443DC244xxx omci ont-lineprofile-id 2 ont-srvprofile-id 2 desc 7424j0000

#quit

#service-port vlan 3606 gpon 0/5/2 ont 129 gemport 1 multi-service user-vlan 10 rx-cttr 6 tx-cttr 6

#service-port vlan 1101 gpon 0/5/2 ont 129 gemport 2 multi-service user-vlan 20 rx-cttr 11 tx-cttr 11

 

-------- Delete service ont -------

#undo service-port port (fr/sl/po) ont (id) gemport (id)

------- Delete ont -------

#ont delete (port id) (onu id)

------- display ont -------

#config

#interface gpon 0/5

#display ont info (port id (0-8)) all      // ดู sn ของแต่ละ port

#display service-port port (fr/sl/po) ont (id)   // ดู rx-index tx-index (download)

#display service-port port (fr/sl/po) ont (port id (0-8)

#undo service-port (port id (0-8)

------- add vlan  uplink -------

#vlan 3600 smart vpi 1 vci 32 rx 6 tx 6

#port vlan 3600 0/2 0

---------------------------- -- Huawei. edit ont-lineprofile -- ---------------------------

#interface gpon 0/5

#display ont info (port id (0-8)) all      // ดู sn ของแต่ละ port

yala(config)#display ont-lineprofile gpon profile-id 12

yala(config)#ont-lineprofile gpon profile-id 12 profile-name P4_InternetUp15M // เข้าไปใน line-profile ที่ต้องการแก้ไข

yala(config-gpon-lineprofile-12)#tcont 1 dba-profile-id 34  // แก้ไข ให้เป็น dba-profile-id 34
